// ✅ enum class — an toàn
enum class Currency { VND, USD, EUR, JPY, GBP };

// Hàm chuyển enum → chuỗi để in ra màn hình
std::string currencyToString(Currency c) {
    switch (c) {
        case Currency::VND: return "VND";
        case Currency::USD: return "USD";
    }
    return "???";
}

// Hàm chuyển chuỗi → enum, ném lỗi nếu không nhận ra
Currency currencyFromString(const std::string& s) {
    if (s == "VND") return Currency::VND;
    if (s == "USD") return Currency::USD;
    throw InvalidInput("Unknown: " + s);
}
💡
Luôn dùng enum class trong C++11 trở lên. enum thường cho phép int x = VND; — compiler không báo lỗi nhưng logic sai.

Operator Overloading
Money.cpp

Dạy C++ cách dùng các ký hiệu + − == >= với kiểu dữ liệu tự tạo.

// Phân tích từng mảnh của dòng đầu — click vào mảnh nào để hiểu:
Money Money::operator+(const Money& o) const {
    assertSameCurrency(*this, o); // kiểm tra cùng đơn vị tiền
    return Money{amount_ + o.amount_, currency_};
    //           ^^^ field của *this  ^^^ field của o (tham số)
}

// So sánh — không dùng == với số thực!
bool Money::operator==(const Money& o) const {
    return currency_ == o.currency_
        && std::fabsl(amount_ - o.amount_) < 1e-9L;
    //   bước 1: cùng đơn vị?  bước 2: gần nhau đủ để coi là bằng?
}
⚠️
Quy tắc vàng: Không dùng == với số thực. 0.1 + 0.2 trong máy tính không bằng 0.3 chính xác do sai số nhị phân. Dùng |a−b| < 1e-9 thay thế.
// Free operator — viết ngoài class, cho phép: account + money
Account& operator+(Account& dst, const Money& m) {
    dst.deposit(m);
    return dst; // trả về chính dst → có thể chain: acc + m1 + m2
}

  std::lock_guard<std::mutex> lk(mtx_);

⚙ PROTECTED — không lock (nội bộ)

void Account::doDeposit(const Money& m) {
  // mutex đã giữ bởi deposit()
  balance_ = Money{
    balance_.amount() + m.amount(),
    balance_.currency()
  };
}
🔒
Nếu cả hai đều lock cùng mutex: deposit() lock → gọi doDeposit() → cũng lock → deadlock ngay. std::mutex không thể lock 2 lần từ cùng 1 thread.

Override — đa hình thực tế

// SavingsAccount: thêm giới hạn rút theo ngày
void SavingsAccount::doWithdraw(const Money& m) override {
    if (withdrawnToday_ + m.amount() > dailyWithdrawCap_)
        throw InsufficientFunds("Vuot han muc rut trong ngay");
    if (balance_.amount() < m.amount())
        throw InsufficientFunds("So du khong du");
    balance_        = Money{balance_.amount() - m.amount(), balance_.currency()};
    withdrawnToday_ += m.amount(); // cộng dồn hôm nay đã rút bao nhiêu
}

// CheckingAccount: cho phép số dư âm đến hạn mức
void CheckingAccount::doWithdraw(const Money& m) override {
    long double after = balance_.amount() - m.amount();
    if (after < -overdraftLimit_)
        throw OverdraftExceeded("Vuot han muc thu chi");
    balance_ = Money{after, balance_.currency()};
}
💬
Cả hai đều override doWithdraw(). Khi Bank gọi qua con trỏ Account*, C++ tự gọi đúng version — đây chính là đa hình (polymorphism).

Giải pháp: lock theo thứ tự cố định

void Bank::transfer(const std::string& srcId,
                    const std::string& dstId,
                    const Money& amount) {

    Account* first  = srcPtr.get();
    Account* second = dstPtr.get();
    if (srcId > dstId) std::swap(first, second);
    //  ^^^ nếu srcId > dstId, đổi chỗ → luôn lock ID nhỏ hơn trước

    std::scoped_lock lock(first->mutex(), second->mutex());
    //  ^^^ lock CẢ HAI cùng lúc, tự mở khi ra scope

    Money converted = converter_.convert(amount, dstPtr->balance_.currency());
    srcPtr->doWithdraw(amount);   // nếu throw → dstPtr chưa bị sửa
    dstPtr->doDeposit(converted); // chỉ chạy nếu rút thành công
}
